import 'dart:io';
import 'dart:convert';
import 'package:flutter/material.dart';
import 'package:image_picker/image_picker.dart';
import 'package:cloud_firestore/cloud_firestore.dart';
import 'package:firebase_auth/firebase_auth.dart';
import 'package:firebase_storage/firebase_storage.dart';
import 'package:http/http.dart' as http;
import 'package:sign_ease/utils/colors_utils.dart';

class SignLangUserChatHead extends StatefulWidget {
  const SignLangUserChatHead({
    Key? key,
    required this.userId,
    required this.userEmail,
    required this.userName,
    required this.userEmails,
  }) : super(key: key);

  final String userId;
  final String userEmail;
  final String userName;
  final List<String> userEmails;

  @override
  State<SignLangUserChatHead> createState() => _SignLangUserChatHeadState();
}

class _SignLangUserChatHeadState extends State<SignLangUserChatHead> {
  final TextEditingController _controller = TextEditingController();
  final ImagePicker _picker = ImagePicker();
  Stream<QuerySnapshot>? _messagesStream;
  String? _chatRoomId;

  @override
  void initState() {
    super.initState();
    _initializeChatRoom();
    _initializeMessagesStream();
  }

  void _initializeChatRoom() {
    final currentUserId = FirebaseAuth.instance.currentUser!.uid;
    final otherUserId = widget.userId;

    _chatRoomId = currentUserId.compareTo(otherUserId) < 0
        ? '${currentUserId}_${otherUserId}'
        : '${otherUserId}_${currentUserId}';

    print('Generated ChatRoomId: $_chatRoomId');
  }

  void _initializeMessagesStream() {
    if (_chatRoomId == null) return;

    try {
      _messagesStream = FirebaseFirestore.instance
          .collection('universalMessages')
          .doc(_chatRoomId)
          .collection('messages')
          .orderBy('timestamp', descending: true)
          .snapshots();

      _messagesStream!.listen((snapshot) {
        for (var doc in snapshot.docs) {
          print('Fetched Message: ${doc.data()}');
        }
      });
    } catch (e) {
      print('Error initializing messages stream: $e');
    }
  }

  Future<void> _saveMessage(String message, String type) async {
    if (_chatRoomId == null) return;

    final messageDoc = FirebaseFirestore.instance
        .collection('universalMessages')
        .doc(_chatRoomId)
        .collection('messages')
        .doc();

    await messageDoc.set({
      'message': message,
      'type': type,
      'timestamp': FieldValue.serverTimestamp(),
      'sender': FirebaseAuth.instance.currentUser!.uid,
    });

    final chatDocRef = FirebaseFirestore.instance
        .collection('universalChats')
        .doc(_chatRoomId);

    await chatDocRef.set({
      'participants': [
        FirebaseAuth.instance.currentUser!.uid,
        widget.userId,
      ],
      'lastMessage': message,
      'lastMessageType': type,
      'timestamp': FieldValue.serverTimestamp(),
    }, SetOptions(merge: true));
  }

  void _sendMessage() {
    final message = _controller.text.trim();
    if (message.isNotEmpty && _chatRoomId != null) {
      _saveMessage(message, 'text').catchError((error) {
        print('Error saving message: $error');
        ScaffoldMessenger.of(context).showSnackBar(
          SnackBar(content: Text('Error sending message: $error')),
        );
      });
      _controller.clear();
    }
  }

  Future<void> _pickImage() async {
    final XFile? image = await _picker.pickImage(source: ImageSource.camera);
    if (image != null) {
      await _uploadAndPredictImage(image);
    }
  }

  Future<void> _uploadAndPredictImage(XFile image) async {
    final File imageFile = File(image.path);

    try {
      final request = http.MultipartRequest(
        'POST',
        Uri.parse('http://your-server-ip:5000/predict'),
      );

      request.files
          .add(await http.MultipartFile.fromPath('image', imageFile.path));
      final response = await request.send();

      if (response.statusCode == 200) {
        final responseData = await http.Response.fromStream(response);
        final predictions = jsonDecode(responseData.body);

        print('Predictions: $predictions');

        // Save the prediction result as a message
        if (predictions['predictions'] != null) {
          _saveMessage(predictions['predictions'].toString(), 'text');
          ScaffoldMessenger.of(context).showSnackBar(
            SnackBar(
                content: Text('Prediction: ${predictions['predictions']}')),
          );
        }
      } else {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Failed to get prediction')),
        );
      }
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Error: $e')),
      );
    }
  }

  Future<void> _deleteMessage(String messageId) async {
    if (_chatRoomId == null) return;

    try {
      await FirebaseFirestore.instance
          .collection('universalMessages')
          .doc(_chatRoomId)
          .collection('messages')
          .doc(messageId)
          .delete();

      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Message deleted successfully')),
      );
    } catch (e) {
      ScaffoldMessenger.of(context).showSnackBar(
        SnackBar(content: Text('Error deleting message: $e')),
      );
    }
  }

  void _confirmDeleteMessage(String messageId) {
    showDialog(
      context: context,
      builder: (context) => AlertDialog(
        title: const Text('Delete Message'),
        content: const Text('Are you sure you want to delete this message?'),
        actions: [
          TextButton(
            onPressed: () {
              Navigator.of(context).pop();
            },
            child: const Text('Cancel'),
          ),
          TextButton(
            onPressed: () {
              _deleteMessage(messageId);
              Navigator.of(context).pop();
            },
            child: const Text('Delete'),
          ),
        ],
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.userName),
        backgroundColor: hexStringToColor("2986cc"),
      ),
      body: Column(
        children: [
          Expanded(
            child: StreamBuilder<QuerySnapshot>(
              stream: _messagesStream,
              builder: (context, snapshot) {
                if (!snapshot.hasData || snapshot.data!.docs.isEmpty) {
                  return const Center(child: Text('No messages yet'));
                }

                final messages = snapshot.data!.docs
                    .map((doc) => doc.data() as Map<String, dynamic>)
                    .toList();

                return ListView.builder(
                  reverse: true,
                  itemCount: messages.length,
                  itemBuilder: (context, index) {
                    final messageData = messages[index];
                    final sender = messageData['sender'];
                    final isCurrentUser =
                        sender == FirebaseAuth.instance.currentUser!.uid;

                    return _buildMessageBubble(
                      messageData['message'],
                      messageData['type'],
                      isCurrentUser,
                      snapshot.data!.docs[index].id,
                    );
                  },
                );
              },
            ),
          ),
          _buildMessageInput(),
        ],
      ),
      backgroundColor: hexStringToColor("ffffff"),
    );
  }

  Widget _buildMessageBubble(
      String message, String type, bool isCurrentUser, String messageId) {
    return GestureDetector(
      onLongPress: () => _confirmDeleteMessage(messageId),
      child: Align(
        alignment: isCurrentUser ? Alignment.centerRight : Alignment.centerLeft,
        child: Container(
          padding: const EdgeInsets.all(12.0),
          margin: const EdgeInsets.symmetric(vertical: 4.0, horizontal: 8.0),
          decoration: BoxDecoration(
            color: isCurrentUser
                ? Colors.greenAccent.shade200
                : Colors.blueAccent.shade100,
            borderRadius: BorderRadius.only(
              topLeft: const Radius.circular(12),
              topRight: const Radius.circular(12),
              bottomLeft:
                  isCurrentUser ? const Radius.circular(12) : Radius.zero,
              bottomRight:
                  isCurrentUser ? Radius.zero : const Radius.circular(12),
            ),
          ),
          child: type == 'text'
              ? Text(message, style: const TextStyle(color: Colors.black))
              : (type == 'image'
                  ? _buildImageMessage(message)
                  : const Text('Unsupported message type')),
        ),
      ),
    );
  }

  Widget _buildImageMessage(String imageUrl) {
    return Image.network(
      imageUrl,
      width: 200,
      height: 200,
      fit: BoxFit.cover,
      loadingBuilder: (context, child, progress) {
        if (progress == null) return child;
        return const Center(child: CircularProgressIndicator());
      },
      errorBuilder: (context, error, stackTrace) {
        return const Center(child: Icon(Icons.error));
      },
    );
  }

  Widget _buildMessageInput() {
    return Padding(
      padding: const EdgeInsets.all(8.0),
      child: Row(
        children: [
          Expanded(
            child: TextField(
              cursorColor: Colors.white,
              style: const TextStyle(color: Colors.black),
              controller: _controller,
              decoration: InputDecoration(
                filled: true,
                fillColor: Colors.white,
                hintText: 'Enter your message',
                hintStyle: const TextStyle(color: Colors.grey),
                border: OutlineInputBorder(
                  borderRadius: BorderRadius.circular(12),
                  borderSide: BorderSide.none,
                ),
              ),
            ),
          ),
          IconButton(
            icon: const Icon(Icons.send, color: Colors.blue),
            onPressed: _sendMessage,
          ),
          IconButton(
            icon: const Icon(Icons.camera_alt, color: Colors.blue),
            onPressed: _pickImage,
          ),
        ],
      ),
    );
  }
}




































































































import 'dart:io';
import 'dart:typed_data';
import 'package:flutter/material.dart';
// ignore: unnecessary_import
import 'package:flutter/services.dart'; // For loading assets
import 'package:image_picker/image_picker.dart';
import 'package:tflite_flutter/tflite_flutter.dart';
import 'package:image/image.dart' as img; // For image preprocessing

class SignLangUserChatHead extends StatefulWidget {
  const SignLangUserChatHead({
    Key? key,
    required this.userId,
    required this.userEmail,
    required this.userName,
    required this.userEmails,
  }) : super(key: key);

  final String userId;
  final String userEmail;
  final String userName;
  final List<String> userEmails;

  @override
  State<SignLangUserChatHead> createState() => _SignLangUserChatHeadState();
}

class _SignLangUserChatHeadState extends State<SignLangUserChatHead> {
  late Interpreter _interpreter;
  bool _isModelLoaded = false;
  List<String> _labels = [];
  String _predictedLabel = "";
  final ImagePicker _picker = ImagePicker();

  @override
  void initState() {
    super.initState();
    _loadModel();
  }

  /// Load the TFLite model and labels
  Future<void> _loadModel() async {
    try {
      _interpreter = await Interpreter.fromAsset('assets/model.tflite');

      String labelsData =
          await DefaultAssetBundle.of(context).loadString('assets/labels.txt');
      _labels = labelsData.split('\n').map((e) => e.trim()).toList();

      setState(() {
        _isModelLoaded = true;
      });

      print("Model and labels loaded successfully!");
    } catch (e) {
      print("Error loading model or labels: $e");
    }
  }

  /// Pick an image using the ImagePicker and run inference
  Future<void> _pickAndPredictImage() async {
    try {
      final XFile? image = await _picker.pickImage(source: ImageSource.gallery);

      if (image != null) {
        final result = await _runInference(File(image.path));
        setState(() {
          _predictedLabel = result;
        });
      }
    } catch (e) {
      print("Error picking image: $e");
    }
  }

  /// Run inference on the selected image
  Future<String> _runInference(File imageFile) async {
    if (!_isModelLoaded) {
      return "Model not loaded";
    }
    try {
      img.Image? image = img.decodeImage(imageFile.readAsBytesSync());
      if (image == null) return "Invalid Image";

      img.Image resizedImage = img.copyResize(image, width: 112, height: 112);
      Float32List input = Float32List(112 * 112 * 3);
      for (int y = 0; y < 112; y++) {
        for (int x = 0; x < 112; x++) {
          int pixel = resizedImage.getPixel(x, y);
          input[(y * 112 + x) * 3 + 0] = ((pixel >> 16) & 0xFF) / 127.5 - 1.0;
          input[(y * 112 + x) * 3 + 1] = ((pixel >> 8) & 0xFF) / 127.5 - 1.0;
          input[(y * 112 + x) * 3 + 2] = (pixel & 0xFF) / 127.5 - 1.0;
        }
      }

      var inputTensor = input.reshape([1, 112, 112, 3]);
      var output =
          List.filled(_labels.length, 0.0).reshape([1, _labels.length]);

      _interpreter.run(inputTensor, output);
      int predictedIndex =
          output[0].indexOf(output[0].reduce((a, b) => a > b ? a : b));
      return _labels[predictedIndex];
    } catch (e) {
      print("Error during inference: $e");
      return "Error";
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text(widget.userName),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            ElevatedButton(
              onPressed: _isModelLoaded ? _pickAndPredictImage : null,
              child: const Text("Select an Image"),
            ),
            const SizedBox(height: 20),
            Text(
              "Prediction: $_predictedLabel",
              style: const TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
            ),
          ],
        ),
      ),
    );
  }
}
